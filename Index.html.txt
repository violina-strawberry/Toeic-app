    <div class="card" id="main-card">
        <div id="mode-label">目前模式：英翻中</div>
        
        <div class="word-box">
            <h1 id="word-display" class="loading">載入中...</h1>
            <div id="ans-display">翻譯載入中</div>
        </div>

        <div class="button-group">
            <button class="primary-btn" onclick="showAns()">查看翻譯</button>
            <button class="secondary-btn" onclick="nextWord()">下一個單字</button>
            <button class="mode-btn" onclick="toggleMode()">切換 英中 / 中英</button>
        </div>
    </div>

    <script>
        let myWords = [];
        let currentIndex = 0;
        let isEnToZh = true;
        const card = document.getElementById('main-card');
        const wordEl = document.getElementById('word-display');
        const ansEl = document.getElementById('ans-display');
        const modeEl = document.getElementById('mode-label');

        // 1. 從 JSON 檔案抓取資料
        async function loadWords() {
            try {
                // 注意：確保你的 words.json 檔案已經上傳到同一個倉庫
                const response = await fetch('words.json');
                if (!response.ok) throw new Error('找不到 words.json 檔案');
                
                myWords = await response.json();
                
                if (myWords.length > 0) {
                    wordEl.classList.remove('loading');
                    nextWord();
                } else {
                    wordEl.innerText = "JSON 檔內沒有單字";
                }
            } catch (error) {
                wordEl.innerText = "讀取失敗";
                ansEl.innerText = "請確認 words.json 是否已上傳";
                ansEl.style.opacity = 1;
                console.error(error);
            }
        }

        // 2. 顯示下一題
        function nextWord() {
            if (myWords.length === 0) return;
            
            // 隨機選一個索引
            currentIndex = Math.floor(Math.random() * myWords.length);
            
            // 隱藏答案動畫
            card.classList.remove('show-answer');
            
            // 更新文字
            setTimeout(updateDisplay, 150); // 微小延遲讓換題感覺更順暢
        }

        // 3. 更新顯示內容
        function updateDisplay() {
            const item = myWords[currentIndex];
            if (isEnToZh) {
                wordEl.innerText = item.en;
                ansEl.innerText = item.zh;
            } else {
                wordEl.innerText = item.zh;
                ansEl.innerText = item.en;
            }
        }

        // 4. 顯示答案
        function showAns() {
            card.classList.add('show-answer');
        }

        // 5. 切換模式
        function toggleMode() {
            isEnToZh = !isEnToZh;
            modeEl.innerText = isEnToZh ? "目前模式：英翻中" : "目前模式：中翻英";
            updateDisplay();
            card.classList.remove('show-answer');
        }

        // 初始化：啟動程式
        loadWords();
    </script>
