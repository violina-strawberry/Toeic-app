<div class="card">
    <div id="mode-label" style="font-size: 12px; color: #999;">目前模式：英翻中</div>
    <div class="word" id="word-display">Loading...</div>
    <div class="ans" id="ans-display">翻譯載入中</div>
    
    <button onclick="showAns()">看答案</button>
    <button onclick="nextWord()">下一題 (隨機)</button>
    <button class="mode-btn" onclick="toggleMode()">切換 英中/中英 模式</button>
</div>

<script>
    // 【重點】在這裡增加你的新單字！
    const myWords = [
        { en: "Accomplish", zh: "完成；實現" },
        { en: "Mandatory", zh: "強制性的" },
        { en: "Incentive", zh: "誘因" },
        { en: "Execute", zh: "執行" },
        { en: "Quarterly", zh: "每季的" }
    ];

    let currentIndex = 0;
    let isEnToZh = true;

    function nextWord() {
        currentIndex = Math.floor(Math.random() * myWords.length);
        document.getElementById('ans-display').style.visibility = "hidden";
        updateDisplay();
    }

    function updateDisplay() {
        const item = myWords[currentIndex];
        if (isEnToZh) {
            document.getElementById('word-display').innerText = item.en;
            document.getElementById('ans-display').innerText = item.zh;
        } else {
            document.getElementById('word-display').innerText = item.zh;
            document.getElementById('ans-display').innerText = item.en;
        }
    }

    function showAns() {
        document.getElementById('ans-display').style.visibility = "visible";
    }

    function toggleMode() {
        isEnToZh = !isEnToZh;
        document.getElementById('mode-label').innerText = isEnToZh ? "目前模式：英翻中" : "目前模式：中翻英";
        updateDisplay();
    }

    // 網頁打開後先顯示第一題
    nextWord();
</script>
